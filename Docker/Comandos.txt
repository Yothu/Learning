docker login						#Logea a el Docker Hub
docker pull [image]:[tag]				#Descarga imagenes docker desde el repositorio Docker Hub
docker push [User]/[image]:[tag]

docker tag [image ID] [User]/[image]:[tag] 		#Tagea una imagen 

docker build [Docker File]				#Contruye DockerFile's a imagenes Docker
docker images						#Muestra imagenes Docker descargadas

docker ps						#Muestra contenedores en funcionamiento
docker ps -a						#Muestra historial de contenedores que estuvieron en funcionamiento 


docker run -d -p 80:80 [image]:[tag]			#Crea y corre una imagen Docker en background (-d) y conectando el puerto 80 de la maquina y el 80 del contenedor 
docker run -it [i]:[t]					#Crea y corre una imagen Docker con terminal interactivo del contenedor creado
docker run -v [path from /home]:[folder] [i]:[t]	#Crea y corre una imagen Docker con informacion guardada previamente

docker exec -it [Container ID] bash			#Accede a la terminal del contenedor

docker stop [C ID]					#Detiene un Contenedor que esta andando
docker kill [C ID]					#Detiene forzosamente un Contenedor
docker start [i]:[t]					#Corre una imagen Docker ya creada

docker commit [C ID] [User]/[image]			#Crea una nuevo imagen Docker basada en un container editado

docker network create [network]				#Crea una network, conexxion entre varios contenedores
